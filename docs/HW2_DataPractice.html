(a) standardizing

std_data <- function(X){
  n = nrow(X)
  p = ncol(X)
  Xbar = as.numeric((t(replicate(n,1)) %*% X) / n)
  Xc = X - t(replicate(n,Xbar))
  S = (t(X) %*% (diag(replicate(n,1)) - matrix(replicate(n*n,1), nrow = n)/n)  %*% X)/(n-1)
  D = diag(1/sqrt(diag(S)))
  Z = Xc %*% D 
  return(Z)
}
